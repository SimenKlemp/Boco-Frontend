<script>
export default {
  name: "StatusSystemMessage",
 
  props: {
    rental: {
      type: Object,
      required: true,
    },
  },
  computed: {
    status() {
      switch (this.rental.status) {
        case "ACCEPTED":
          return "Akseptert";
        case "ACTIVE":
          return "Aktiv";
        case "PENDING":
          return "Avventer";
        case "REJECTED":
          return "Avvist";
        case "CANCELED":
          return "Kansellert";
        case "FINISHED":
          return "Avsluttet";
        default:
          return "Default";
      }
    },
  },
};
</script>
